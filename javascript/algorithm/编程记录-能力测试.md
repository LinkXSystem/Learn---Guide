> 数组求和
```
# 递归式编程
function sum(array) {
    let len = array.length;
    if(len == 0){
        return 0;
    } else if (len == 1){
        return array[0];
    } else {
        return array[0] + sum(array.slice(1));
    }
}
# 常规式编程
function sum(array) {
    if (array) return 0;
    let count = 0;
    array.forEach(function(num) {
        count += num;
    });
    return count;
}
# eval式编程
function sum(array) {
    return eval(array.join('+'));
}

# 函数式编程
function sum(array) {
    return array.reduce(function(prev, curr, index, array) {
        return prev + curr;
    });
}
```

function remove(array, item) {
    let cache = array.slice(0);
    for (let i = 0; i < store.length; i++) {
        if (store[i] == item) {
            cache.splice(i, 1);
            i--;
        }
    }
    return cache;
}
# 使用 filter 方法
function remove(array, item) {
    return array.filter(function(num) {
        return num !== item;
    });
}
