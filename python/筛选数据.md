> list [列表]
```
from random import randint

data = [randint(-10, 10) for _ in range(10)]

store = filter(lambda x: x >= 0, data);

list(store)

<!-- 解析式写法 -->
[x for x in data if x >= 0]

import timeit

time = timeit.Timer("[x for x in data if x >= 0]", 
    "data=[randint(-10, 10) for _ in range(10)]")

time.timeit()
```
